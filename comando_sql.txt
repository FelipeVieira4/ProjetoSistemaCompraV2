create table produto(
	id int primary key auto_increment,
	preco float,
	nome varchar(255),
	caminhoFoto varchar(255),
	contatoDistribuidora varchar(255),
	localproduzido varchar(255),
	descricao varchar(255)
);

create table user_info(
	id int primary key auto_increment,
	nome varchar(45),
	senha varchar(255)
);
